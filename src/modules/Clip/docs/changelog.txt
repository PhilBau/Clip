        list:

            $core_titlefield => $pubtype.titlefield
        display:
            Publication data embeded in the $pubdata array

            $core_uniqueid => $pubdata.core_uniqueid
            $core_creator => $pubdata.core_creator
            $core_approvalstate => $pubdata.core_approvalstate
